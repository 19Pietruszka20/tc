module zad2(a,b);
input [7:0]a;
output reg [7:0]b;
integer i;
always@(*)
if(a[7]==0) begin
b=a;
end else begin
for(i=0; i<8; i=i+1) begin
b[i]=~a[i];
end
for(i=0; i<8; i=i+1) begin
if(b[i]==0) begin
b[i]=b[i]+1;
end else begin
b[i]=b[i]-1;
end
end
end
endmodule
